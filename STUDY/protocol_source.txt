{

	m_pServerSoc->bindRecvCallback(std::bind(&CStatusHandler::recvHandler, this, std::placeholders::_1, std::placeholders::_2));

	m_pServerSoc->setMaxClient(gpParam->getIntOpt("STATUSHND_CLIENT_MAX"));

	std::this_thread::sleep_for(std::chrono::milliseconds(10));









	return true;

}



void CStatusHandler::stop()

{

	this->savePersistanceFile();



	if (!m_bIsStarted)

		return;



	if (m_pServerSoc != NULL)

	{

		m_pServerSoc->stop();

		delete m_pServerSoc;

		m_pServerSoc = NULL;

	}





	if (m_pThreadMES != NULL && m_bMESThreadRun == true)

	{

		m_bMESThreadRun = false;

		m_pThreadMES->join();

		delete m_pThreadMES;

		m_pThreadMES = NULL;

		



	}





	m_sharedMem.remove();

}

bool CStatusHandler::readPersistanceFile()

{

	FILE* fp;

	char szFile[256] = { 0, }, buf[1024] = { 0, };

	int nChIndex = 0;

	unsigned int nVal = 0;

	std::string strLine;



	sprintf(szFile, "%s/%s", gszHomeDir, PERSISTANCE_FILE);

	fp = fopen(szFile, "r");



	if (fp == nullptr)

	{

		writeSystemError("[CStatusHandler] persistance file can't read.");

		return false;

	}



	while (!feof(fp))

	{

		strLine.clear();



		fgets(buf, sizeof(buf), fp);

		strLine.append(buf);



		trim(strLine);



		if (strLine[0] != '#' && strLine[0] != '¥r' && strLine[0] != '¥n' && strLine.size() > (size_t)0)

		{

			vector<string> tokens;

			stringstream sstream(strLine);

			string strTemp;



			while (getline(sstream, strTemp, '¥t'))

			{

				tokens.push_back(strTemp);

			}



			if (tokens.size() < (size_t)1)

				continue;



			if (tokens[0].find("DIN_", 0) != std::string::npos)

			{

				nChIndex = atoi(tokens[0].substr(tokens[0].find("_", 0) + 1).c_str());

				nVal = (unsigned int)atoi(tokens[1].c_str());

				m_Status.anIOInputCount[nChIndex] = nVal;

			}

		}

	}



	fclose(fp);



	return true;

}

	m_Status.windowClientCount = (unsigned char)gpParam->getCurrentConnectedClient();

	// Serial魹ｽ魹ｽﾄ｡ 魹ｽﾌｸ魹ｽ 魹ｽ魹ｽ魹ｽ魹ｽ

			strcpy(m_Status.aszSerialDevName[nPort][nSlave], gpParam->getStrOpt(buf));

		} // for SERIAL_DEVCNT_%d

	// Socket 魹ｽ魹ｽﾄ｡ 魹ｽﾌｸ魹ｽ 魹ｽ魹ｽ魹ｽ魹ｽ

void CStatusHandler::recvHandler(CEVServerSocketWorker* pWorker, ssize_t nLen)

{

	char buf[SOCKET_BUFFER_SIZE] = { 0, };

	int nRecvLen = 0, i;

	char cLastChr = 0x00;// , cCurChr = 0x00;

	char cIsLastEtx = 0x00;

	char szJobType[16] = { 0, };



	if (pWorker == NULL) return;

	if (nLen <= 0) return;



	std::this_thread::sleep_for(std::chrono::milliseconds(10));





	m_mutexRecvBuf.lock();





	// 魹ｽ魹ｽ魹ｽﾅｵ魹ｽ 魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ ﾃｳ魹ｽ魹ｽ

	pWorker->getRecvData(buf, sizeof(buf), nRecvLen);

	cLastChr = buf[nRecvLen - 1];

	cIsLastEtx = buf[nRecvLen - 3];



	for (i = 0; i < nRecvLen; i++)

		m_dequeRecv.push_back(buf[i]);





	//if (cLastChr != ASCII_ETX && cLastChr != '}')

	if (cIsLastEtx != ASCII_ETX && cLastChr != '}')

	{

		m_mutexRecvBuf.unlock();

		return;

	}



	// 魹ｽ魹ｽ魹ｽ魹ｽ 魹ｽ魹ｽ魹ｽﾊｿ魹ｽ STX 魹ｽﾇｴ魹ｽ "{"魹ｽ魹ｽ 魹ｽﾆｴﾏｶ魹ｽ魹ｽ 魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ 魹ｽ魹ｽ魹ｽ魹ｽ 魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ.

	while ((m_dequeRecv.front() != ASCII_STX && m_dequeRecv.front() != '{') && !m_dequeRecv.empty())

	{

		m_dequeRecv.pop_front();

	}





	if (m_dequeRecv.size() > 2)

	{

		

		int nProtocolType = checkRequestProtocol(pWorker, szJobType);

		





		if (nProtocolType == 0)

		{	// ASCII 魹ｽ魹ｽ魹ｽ

			processSocketASCII(pWorker, nLen);

		}

		else if (nProtocolType == 1)

		{	// Binary 魹ｽ魹ｽ魹ｽ

			processSocketBinary(pWorker, nLen);

		}

		else if (nProtocolType == 2)

		{	// Json 魹ｽ魹ｽ魹ｽ

			processSocketJson(pWorker, nLen, szJobType);

		}		



	}



	m_mutexRecvBuf.unlock();



}





int CStatusHandler::checkRequestProtocol(CEVServerSocketWorker* pWorker, char* szJobType)

{

	//int nProtocolType = 0;







	if (m_dequeRecv[0] == ASCII_STX)

	{

		if (m_dequeRecv[1] == 'a')

		{

			//nProtocolType = 0;		// ASCII 魹ｽ魹ｽ魹ｽ

			return 0;

		}

		else if (m_dequeRecv[1] == 'b')

		{

			//nProtocolType = 1;		// Binary 魹ｽ魹ｽ魹ｽ

			return 1;

		}		

		else

			return -1;

	}

	else if (m_dequeRecv[0] == '{')

	{

		std::istringstream streamRecv;

		std::string strRecv, strSend;

		Json::CharReaderBuilder builder;

		Json::Value main;

		JSONCPP_STRING jsonErr;



		builder["collectComments"] = false;



		while (!m_dequeRecv.empty())

		{

			strRecv += m_dequeRecv.front();

			m_dequeRecv.pop_front();

		}



		streamRecv.str(strRecv);	// 魹ｽ魹ｽ魹ｽ魹ｽ 魹ｽ魹ｽ魹ｽ魹ｽ魹ｽﾍｸ魹ｽ Stream魹ｽ魹ｽ 魹ｽﾖｴﾂｴ魹ｽ.



		//////////////////////////////////////////////////////////////////

		// Parsing魹ｽ魹ｽ 魹ｽ魹ｽﾈｿ魹ｽ魹ｽ ﾃｼﾅｩ

		if (!Json::parseFromStream(builder, streamRecv, &main, &jsonErr))			// Parsing

		{

			Json::StreamWriterBuilder writerBuilder;



			writerBuilder["commentStyle"] = "None";

			writerBuilder["indentation"] = "¥t";		// Tab

			writerBuilder["useSpecialFloats"] = false;

			writerBuilder["precision"] = 5;				// 魹ｽﾒｼ魹ｽ魹ｽ魹ｽ



			// ERRORﾃｳ魹ｽ魹ｽ

			main.clear();



			main["protocol"] = "ms";

			main["direction"] = "response";

			main["errno"] = 1;

			main["errtxt"] = "Invalid Data. Json data can't to parse.";



			strSend = Json::writeString(writerBuilder, main);

			pWorker->send(strSend.c_str(), strSend.size());



			return -1;

		}

		else

		{

			bool bCorrect = true;



			if(main["protocol"].empty() || main["direction"].empty() || main["job"].empty())

				bCorrect = false;



			if (bCorrect)

			{

				if (strcmp(main["protocol"].asCString(), "ms") != 0 ||

					strcmp(main["direction"].asCString(), "request") != 0

					)

					bCorrect = false;

			}



			if (!bCorrect)

			{

				if (strcmp(main["protocol"].asCString(), "ms") != 0 ||

					strcmp(main["direction"].asCString(), "request") != 0)

				{

					Json::StreamWriterBuilder writerBuilder;



					writerBuilder["commentStyle"] = "None";

					writerBuilder["indentation"] = "¥t";		// Tab

					writerBuilder["useSpecialFloats"] = false;

					writerBuilder["precision"] = 5;				// 魹ｽﾒｼ魹ｽ魹ｽ魹ｽ





					// ERRORﾃｳ魹ｽ魹ｽ

					main.clear();



					main["protocol"] = "ms";

					main["direction"] = "response";

					main["errno"] = 2;

					main["errtxt"] = "Protocol not matched.(protocol, direction)";



					strSend = Json::writeString(writerBuilder, main);

					pWorker->send(strSend.c_str(), strSend.size());



					return -1;

				}

			}

		}



		strcpy(szJobType, main["job"].asCString());



		return 2;



	}

	else

		return -1;



	return -1;

}

{

	int nRecvLen = 0, i;

	bool bExistSTX = false, bExistETX = false;

	char sCRC[3] = { 0, }, sProtocol[128] = { 0, }, szKey[64] = { 0, }, szEucKr[64] = { 0, };

	char szUS[2] = { ASCII_US, 0 };

	unsigned short usCRC = 0;

	std::string strOneProtocol, strOrgProtocol, strSend;

	std::string strJobType;

	int nRtn = 0, nIndex = 0;

	int nPort = 0, nSlave = 0;

	

	int nSlaveCnt = 0, n485PortCnt = 0;







	nRecvLen = (int)m_dequeRecv.size();





	for (i = 0; i < nRecvLen; i++)

	{

		strOrgProtocol += m_dequeRecv.front();		// CRC魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ 魹ｽ魹ｽ魹ｽ魹ｽ <STX><ETX>魹ｽ魹ｽ魹ｽﾅｵ魹ｽ魹ｽ魹ｽ 魹ｽ魹ｽ魹ｽ魹ｽ 魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ 魹ｽ魹ｽ魹ｽ魹ｽ魹ｽﾑｴ魹ｽ.



		if (m_dequeRecv.front() == ASCII_STX)

		{			

			m_dequeRecv.pop_front();				// STX魹ｽ魹ｽ 魹ｽ魹ｽ魹ｽ魹ｽ魹ｽﾑｴ魹ｽ.

			bExistSTX = true;

			continue;

		}



		if (m_dequeRecv.front() == ASCII_ETX)

		{

			m_dequeRecv.pop_front();				// ETX魹ｽ魹ｽ魹ｽ魹ｽ 魹ｽ魹ｽ 魹ｽ魹ｽ魹ｽﾄｴ魹ｽ 魹ｽ魹ｽ魹ｽ魹ｽ魹ｽﾑｴ魹ｽ.

			bExistETX = true;

			break;

		}



		strOneProtocol += m_dequeRecv.front();		// strOneProtocol魹ｽ魹ｽ <STX>魹ｽ魹ｽ <ETX>魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ 魹ｽ魹ｽ魹ｽﾚｿ魹ｽ(STX,ETX魹ｽ魹ｽ魹ｽ魹ｽ)

	}



	// CRCﾃｼﾅｩ

	bool bCRCSuccess = false;



	if (m_dequeRecv.size() >= 2)

	{

		sCRC[0] = m_dequeRecv.front();		m_dequeRecv.pop_front();

		sCRC[1] = m_dequeRecv.front();		m_dequeRecv.pop_front();



		usCRC = gpParam->getCRC16(0, (unsigned char*)strOrgProtocol.c_str(), (int)strOrgProtocol.size());



		if (((usCRC & 0xFF00) >> 8) == sCRC[1] && (usCRC & 0x00FF) == sCRC[0])

			bCRCSuccess = true;

	}



	if (bCRCSuccess == false)

	{

		sprintf(sProtocol, "%ca%cerror%c%d%c", ASCII_STX, ASCII_GS, ASCII_US, 99, ASCII_ETX);

		pWorker->send(sProtocol, strlen(sProtocol));



		return;

	}



	



	if (bExistSTX && bExistETX && strOneProtocol[0] == 'a' && strOneProtocol[1] == ASCII_GS)

	{

		std::stringstream ss(strOneProtocol);

		std::string token, strKey, strValue;



		char cDelim = '.';



		while (std::getline(ss, token, cDelim))

			vTokens.push_back(token);



		if (vTokens.size() >= 3 || (vTokens[2] == "485" && vTokens.size() >= 4))

					return;

				nSlave = (int)atoi(vTokens[4].c_str());

			}





			////////////////////////////////

			// Header

			sprintf(sProtocol, "%cms.res.%s%c", ASCII_STX, strJobType.c_str(), ASCII_GS);

			////////////////////////////////

				if (strJobType == "485all" || strJobType == "485")

					strSend.append(szUS);					// US



					sprintf(szKey, "DIO_OUT_NAME_%d", i);

					memset(szEucKr, 0, sizeof(szEucKr));

					CEVUtils::evUtf8ToEucKr((char*)gpParam->getStrOpt(szKey), szEucKr, sizeof(szEucKr));		// 魹ｽﾑｱ魹ｽ ﾄｳ魹ｽ魹ｽ魹ｽﾍｼ魹ｽ 魹ｽ魹ｽﾈｯ

					strSend.append(szEucKr);		//strSend.append(gpParam->getStrOpt(szKey));

					strSend.append(szUS);					// US



					sprintf(sProtocol, "%d", (int)m_Status.abIOOutputStatus[i]);

					strSend.append(sProtocol);				// Value

					strSend.append(szUS);					// US



					strSend.append("99");					// Unit

					strSend.append(szUS);					// US



					strSend.append("0");					// Min Value

					strSend.append(szUS);					// US



					strSend.append("0");					// Max Value



					sprintf(sProtocol, "%c", ASCII_RS);

					strSend.append(sProtocol);				// RS

				}





				//main["dio"].setComment("// DIO", Json::commentBefore);

			}







			/////////////////////////////

			// Tail

			sprintf(sProtocol, "%c", ASCII_ETX);

			strSend.append(sProtocol);					// ETX



			/*

			/////////////////////////////

			// 魹ｽﾑｱﾛｶ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ UTF-8 -> EUC-KR魹ｽ魹ｽ 魹ｽ魹ｽﾈｯ魹ｽﾛｾ魹ｽ 魹ｽ魹ｽ魹ｽ魹ｽ ->20230907 魹ｽ魹ｽ魹ｽ魹ｽ. 魹ｽ魹ｽ NAME魹ｽ魹ｽ魹ｽ魹ｽ 魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ 魹ｽ魹ｽ魹ｽ魹ｽ

			char* pBufConv = nullptr;

			nLen = strSend.size();



			pBufConv = new char[nLen];

			memset(pBufConv, 0, nLen);

			//memcpy(pBufConv, strSend.c_str(), strSend.size());



			CEVUtils::evUtf8ToEucKr((char*)strSend.c_str(), pBufConv, nLen);

			

			strSend.clear();			

			strSend.append(pBufConv);



			delete[] pBufConv;

			*/



			/////////////////////////////

			// CRC

			usCRC = gpParam->getCRC16(0, (unsigned char*)strSend.c_str(), (int)strSend.size());

			//usCRC = gpParam->getCRC16(0, (unsigned char*)pBufConv, nLen);



			sCRC[0] = (unsigned char)(usCRC & 0x00FF);							// CRC(LO)

			sCRC[1] = (unsigned char)((usCRC & 0xFF00) >> 8);				// CRC(HI)

			sCRC[2] = 0;

			strSend.append(sCRC);







			pWorker->send(strSend.c_str(), strSend.size());





			nRtn = 1;

		}

		else

			nRtn = 0;

	}

	

	if(nRtn == 0)

	{

		sprintf(sProtocol, "%ca%cerror%c%d%c", ASCII_STX, ASCII_GS, ASCII_US, 1, ASCII_ETX);	// Invalid Protocol

		pWorker->send(sProtocol, strlen(sProtocol));

	}



}









// 魹ｽﾙｸ魹ｽ 魹ｽ魹ｽ魹ｽﾎｱﾗｷ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ 魹ｽ魹ｽﾃｻ魹ｽﾇｴ魹ｽ 魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ 魹ｽ魹ｽﾃｻ魹ｽ魹ｽﾈ｣ 魹ｽ魹ｽ魹ｽ魹ｽ(Binary 魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ 魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ ﾃｳ魹ｽ魹ｽ)

void CStatusHandler::processSocketBinary(CEVServerSocketWorker* pWorker, ssize_t nLen)

{

	int nRecvLen = 0, i;

	bool bExistSTX = false, bExistETX = false;

	char sCRC[3] = { 0, }, sProtocol[128] = { 0, }, szKey[64] = { 0, }, szEucKr[64] = { 0, };

	//char szUS[2] = { ASCII_US, 0 };

	unsigned short usCRC = 0;

	std::string strOneProtocol, strOrgProtocol;// , strSend;

	std::string strJobType;

	int nRtn = 0, nIndex = 0;

	int nPort = 0, nSlave = 0;



	int nSlaveCnt = 0, n485PortCnt = 0;



	std::vector<StructBinarySend> vSend;

	StructBinarySend structBinary;





	nRecvLen = (int)m_dequeRecv.size();





	for (i = 0; i < nRecvLen; i++)

	{

		strOrgProtocol += m_dequeRecv.front();		// CRC魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ 魹ｽ魹ｽ魹ｽ魹ｽ <STX><ETX>魹ｽ魹ｽ魹ｽﾅｵ魹ｽ魹ｽ魹ｽ 魹ｽ魹ｽ魹ｽ魹ｽ 魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ 魹ｽ魹ｽ魹ｽ魹ｽ魹ｽﾑｴ魹ｽ.



		if (m_dequeRecv.front() == ASCII_STX)

		{

			m_dequeRecv.pop_front();				// STX魹ｽ魹ｽ 魹ｽ魹ｽ魹ｽ魹ｽ魹ｽﾑｴ魹ｽ.

			bExistSTX = true;

			continue;

		}



		if (m_dequeRecv.front() == ASCII_ETX)

		{

			m_dequeRecv.pop_front();				// ETX魹ｽ魹ｽ魹ｽ魹ｽ 魹ｽ魹ｽ 魹ｽ魹ｽ魹ｽﾄｴ魹ｽ 魹ｽ魹ｽ魹ｽ魹ｽ魹ｽﾑｴ魹ｽ.

			bExistETX = true;

			break;

		}



		strOneProtocol += m_dequeRecv.front();		// strOneProtocol魹ｽ魹ｽ <STX>魹ｽ魹ｽ <ETX>魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ 魹ｽ魹ｽ魹ｽﾚｿ魹ｽ(STX,ETX魹ｽ魹ｽ魹ｽ魹ｽ)

		m_dequeRecv.pop_front();

	}



	// CRCﾃｼﾅｩ

	bool bCRCSuccess = false;



	if (m_dequeRecv.size() >= 2)

	{

		sCRC[0] = m_dequeRecv.front();		m_dequeRecv.pop_front();

		sCRC[1] = m_dequeRecv.front();		m_dequeRecv.pop_front();



		usCRC = gpParam->getCRC16(0, (unsigned char*)strOrgProtocol.c_str(), (int)strOrgProtocol.size());



		if (((usCRC & 0xFF00) >> 8) == sCRC[1] && (usCRC & 0x00FF) == sCRC[0])

			bCRCSuccess = true;

	}



	if (bCRCSuccess == false)

	{

		sprintf(sProtocol, "%cb%cerror%c%d%c", ASCII_STX, ASCII_GS, ASCII_US, 99, ASCII_ETX);

		pWorker->send(sProtocol, strlen(sProtocol));



		return;

	}







	if (bExistSTX && bExistETX && strOneProtocol[0] == 'b' && strOneProtocol[1] == ASCII_GS)

	{

		std::vector<string> vTokens;

		std::stringstream ss(strOneProtocol);

		std::string token, strKey, strValue;



		char cDelim = '.';



		while (std::getline(ss, token, cDelim))

			vTokens.push_back(token);



		if (vTokens.size() >= 3 || (vTokens[2] == "485" && vTokens.size() >= 4))

		{

			strJobType = vTokens[2];



			if (strJobType == "485all" || strJobType == "485" || strJobType == "232" || strJobType == "soc")

			{

				if (vTokens.size() < 4)

				{

					sprintf(sProtocol, "%cb%cerror%c%d%c", ASCII_STX, ASCII_GS, ASCII_US, 1, ASCII_ETX);

					pWorker->send(sProtocol, strlen(sProtocol));

					return;

				}





				nPort = (int)atoi(vTokens[3].c_str());

			}



			if (strJobType == "485")

			{

				if (vTokens.size() < 5)

				{

					sprintf(sProtocol, "%cb%cerror%c%d%c", ASCII_STX, ASCII_GS, ASCII_US, 1, ASCII_ETX);

					pWorker->send(sProtocol, strlen(sProtocol));

					return;

				}



				nSlave = (int)atoi(vTokens[4].c_str());

			}





			////////////////////////////////

			// Header (魹ｽ魹ｽ魹ｽ魹ｽ 魹ｽ魹ｽ魹ｽﾟｿ魹ｽ 魹ｽ魹ｽ魹ｽ魹ｽ)

			//sprintf(sProtocol, "%cms.res.%s%c", ASCII_STX, strJobType.c_str(), ASCII_GS);

			//strSend.append(sProtocol);







			////////////////////////////////

			// 485 魹ｽﾛｼ魹ｽ

			if (strJobType == "all" || (int)strJobType.find("485", 0) >= 0)

			{

				n485PortCnt = 4;//gpParam->getIntOpt("SERIAL_COUNT") == 8 ? 4 : 2;	 // 魹ｽﾏｴ魹ｽ 魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ 4魹ｽ魹ｽ. 魹ｽ魹ｽ魹ｽﾎｿ魹ｽ魹ｽ魹ｽ 魹ｽ魹ｽ歖ｩ魹ｽﾎｷ魹ｽ 魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ 魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ 魹ｽ魹ｽ魹ｽ魹ｽ 魹ｽ魹ｽ魹ｽ魹ｽ





				if (strJobType == "485all" || strJobType == "485")

				{	// ﾆｯ魹ｽ魹ｽ 魹ｽ魹ｽﾆｮ 1魹ｽ魹ｽ					

					n485PortCnt = nPort + 1;

				}

				else

					nPort = 0;





				for (/*nPort = 0*/; nPort < n485PortCnt; nPort++)

				{

					sprintf(szKey, "SERIAL_USE_%d", nPort);

					if (gpParam->getIntOpt(szKey) == 0) continue;		// 魹ｽ魹ｽ歖ｩ魹ｽ魹ｽ ﾃｼﾅｩ



					nIndex = 0;





					sprintf(szKey, "SERIAL_DEVCNT_%d", nPort);

					nSlaveCnt = gpParam->getIntOpt(szKey);





					if (strJobType == "485")

					{	// ﾆｯ魹ｽ魹ｽ Slave 1魹ｽ魹ｽ

						nSlaveCnt = nSlave + 1;

					}

					else

						nSlave = 0;





					for (/*nSlave = 0*/; nSlave < nSlaveCnt; nSlave++)

					{

						nIndex++;		// 魹ｽﾘｿ魹ｽ 魹ｽ魹ｽ歖ｩ魹ｽﾎｵ魹ｽ魹ｽ魹ｽ continue魹ｽ魹ｽ 魹ｽ魹ｽ魹ｽﾌｹﾇｷ魹ｽ 魹ｽﾏｴ魹ｽ 魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ 魹ｽ魹ｽ魹ｽ魹ｽ(魹ｽ魹ｽ魹ｽ魹ｽ  -1 魹ｽﾘｼ魹ｽ 魹ｽ魹ｽ魹ｽ魹ｽﾒｰ魹ｽ-Zero Base)



						sprintf(szKey, "SERIAL_DEV_USE_%d_%d", nPort, nSlave);

						if (gpParam->getIntOpt(szKey) == 0) continue;		// 魹ｽﾘｴ魹ｽ Slave 魹ｽ魹ｽ歖ｩ魹ｽ魹ｽ ﾃｼﾅｩ



						

						memset(&structBinary, 0, sizeof(StructBinarySend));



						structBinary.cType = 0;							// Type : 485

						structBinary.cMSIndex = (char)(nIndex - 1);		// MS Index

						structBinary.cIndex1 = (char)nPort;				// Port

						structBinary.cIndex2 = (char)nSlave;			// nSlave



						sprintf(szKey, "SERIAL_NAME_%d_%d", nPort, nSlave);

						memset(szEucKr, 0, sizeof(szEucKr));

						CEVUtils::evUtf8ToEucKr((char*)gpParam->getStrOpt(szKey), szEucKr, sizeof(szEucKr));		// 魹ｽﾑｱ魹ｽ ﾄｳ魹ｽ魹ｽ魹ｽﾍｼ魹ｽ 魹ｽ魹ｽﾈｯ

						strcpy(structBinary.szDevName, szEucKr);



						structBinary.fValue = m_Status.afSerialDevValue[nPort][nSlave];	// Value (Float)



						sprintf(szKey, "SERIAL_UNIT_%d_%d", nPort, nSlave);

						structBinary.cUnit = (char)gpParam->getIntOpt(szKey);	// Unit



						sprintf(szKey, "SERIAL_MINVALUE_%d_%d", nPort, nSlave);

						structBinary.fMinValue = (float)gpParam->getDblOpt(szKey);	// Min Value



						sprintf(szKey, "SERIAL_MAXVALUE_%d_%d", nPort, nSlave);

						structBinary.fMaxValue = (float)gpParam->getDblOpt(szKey);	// Min Value



						structBinary.cRS = ASCII_RS;



						vSend.push_back(structBinary);



					}



				}



			} // if (strJobType == "all" || strJobType == "485")





			////////////////////////////////

			// 232 魹ｽﾛｼ魹ｽ

			if (strJobType == "all" || (int)strJobType.find("232") >= 0)

			{

				int n232PortCnt = 4;



				if (strJobType == "232all")

				{

					nPort = 0;

					n232PortCnt = 4;

				}

				else if (strJobType == "232")

				{

					n232PortCnt = nPort + 1;

				}

				else

					nPort = 0;



				for (/*nPort = 0*/; nPort < n232PortCnt; nPort++)

				{

					sprintf(szKey, "SERIAL_USE_%d", nPort + 4);			// RS232魹ｽ魹ｽ 4‾7魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ

					if (gpParam->getIntOpt(szKey) == 0) continue;		// 魹ｽ魹ｽ歖ｩ魹ｽ魹ｽ ﾃｼﾅｩ





					memset(&structBinary, 0, sizeof(StructBinarySend));



					structBinary.cType = 1;							// Type : 232

					structBinary.cMSIndex = (char)0;				// MS Index

					structBinary.cIndex1 = (char)nPort;				// Port

					structBinary.cIndex2 = (char)0;					// nSlave



					sprintf(szKey, "SERIAL_NAME_%d_%d", nPort + 4, 0);

					memset(szEucKr, 0, sizeof(szEucKr));

					CEVUtils::evUtf8ToEucKr((char*)gpParam->getStrOpt(szKey), szEucKr, sizeof(szEucKr));		// 魹ｽﾑｱ魹ｽ ﾄｳ魹ｽ魹ｽ魹ｽﾍｼ魹ｽ 魹ｽ魹ｽﾈｯ

					strcpy(structBinary.szDevName, szEucKr);



					structBinary.fValue = m_Status.afSerialDevValue[nPort + 4][0];	// Value (Float)



					sprintf(szKey, "SERIAL_UNIT_%d_%d", nPort + 4, 0);

					structBinary.cUnit = (char)gpParam->getIntOpt(szKey);	// Unit



					sprintf(szKey, "SERIAL_MINVALUE_%d_%d", nPort + 4, 0);

					structBinary.fMinValue = (float)gpParam->getDblOpt(szKey);	// Min Value



					sprintf(szKey, "SERIAL_MAXVALUE_%d_%d", nPort + 4, 0);

					structBinary.fMaxValue = (float)gpParam->getDblOpt(szKey);	// Min Value



					structBinary.cRS = ASCII_RS;



					vSend.push_back(structBinary);

				}



			} // if (strJobType == "all" || (int)strJobType.find("232") >= 0)





			/////////////////////////////

			// Socket魹ｽﾛｼ魹ｽ

			if (strJobType == "all" || (int)strJobType.find("soc") >= 0)

			{

				int nSocPortCnt = SOCKET_DEV_COUNT;



				if (strJobType == "socall")

				{

					nPort = 0;

					nSocPortCnt = SOCKET_DEV_COUNT;

				}

				else if (strJobType == "soc")

				{

					nSocPortCnt = nPort + 1;

				}

				else

					nPort = 0;



				for (/*nPort = 0*/; nPort < nSocPortCnt; nPort++)

				{

					sprintf(szKey, "SOCKET_USE_%d", nPort);

					if (gpParam->getIntOpt(szKey) == 0) continue;		// 魹ｽ魹ｽ歖ｩ魹ｽ魹ｽ ﾃｼﾅｩ





					memset(&structBinary, 0, sizeof(StructBinarySend));



					structBinary.cType = 2;							// Type : Socket

					structBinary.cMSIndex = (char)0;				// MS Index

					structBinary.cIndex1 = (char)nPort;				// Port

					structBinary.cIndex2 = (char)0;					// nSlave



					sprintf(szKey, "SOCKET_NAME_%d", nPort);

					memset(szEucKr, 0, sizeof(szEucKr));

					CEVUtils::evUtf8ToEucKr((char*)gpParam->getStrOpt(szKey), szEucKr, sizeof(szEucKr));		// 魹ｽﾑｱ魹ｽ ﾄｳ魹ｽ魹ｽ魹ｽﾍｼ魹ｽ 魹ｽ魹ｽﾈｯ

					strcpy(structBinary.szDevName, szEucKr);



					structBinary.nValue = m_Status.anSocketDevValue[nPort];	// Value (Int)



					structBinary.cUnit = (char)99;							// Unit

										

					structBinary.fMinValue = 0.f;							// Min Value

					structBinary.fMaxValue = 0.f;							// Min Value



					structBinary.cRS = ASCII_RS;



					vSend.push_back(structBinary);



				}

			}





			/////////////////////////////

			// DIO

			if (strJobType == "all" || strJobType == "dio")

			{

				// DIO (Input)

				for (i = 0; i < GPIO_IN_COUNT; i++)

				{



					memset(&structBinary, 0, sizeof(StructBinarySend));



					structBinary.cType = 3;							// Type : DIO

					structBinary.cMSIndex = (char)0;				// MS Index

					structBinary.cIndex1 = (char)0;					// Input = 0, Output = 1

					structBinary.cIndex2 = (char)i;					// Channel



					sprintf(szKey, "DIO_IN_NAME_%d", i);

					memset(szEucKr, 0, sizeof(szEucKr));

					CEVUtils::evUtf8ToEucKr((char*)gpParam->getStrOpt(szKey), szEucKr, sizeof(szEucKr));		// 魹ｽﾑｱ魹ｽ ﾄｳ魹ｽ魹ｽ魹ｽﾍｼ魹ｽ 魹ｽ魹ｽﾈｯ

					strcpy(structBinary.szDevName, szEucKr);



					sprintf(szKey, "DIO_IN_ISCOUNT_%d", i);

					if (gpParam->getIntOpt(szKey) == 0)

						structBinary.nValue = m_Status.abIOInputStatus[i];		// Input魹ｽ檥ｵ魹ｽ魹ｽ 魹ｽ魹ｽ魹ｽ

					else

						structBinary.nValue = m_Status.anIOInputCount[i];		// Counter魹ｽ檥ｵ魹ｽ魹ｽ 魹ｽ魹ｽ魹ｽ

									



					structBinary.cUnit = (char)99;							// Unit



					structBinary.fMinValue = 0.f;							// Min Value

					structBinary.fMaxValue = 0.f;							// Min Value



					structBinary.cRS = ASCII_RS;



					vSend.push_back(structBinary);



				}



				// DIO (Output)

				for (i = 0; i < 8; i++)

				{

					memset(&structBinary, 0, sizeof(StructBinarySend));



					structBinary.cType = 3;							// Type : DIO

					structBinary.cMSIndex = (char)0;				// MS Index

					structBinary.cIndex1 = (char)1;					// Input = 0, Output = 1

					structBinary.cIndex2 = (char)i;					// Channel



					sprintf(szKey, "DIO_OUT_NAME_%d", i);

					memset(szEucKr, 0, sizeof(szEucKr));

					CEVUtils::evUtf8ToEucKr((char*)gpParam->getStrOpt(szKey), szEucKr, sizeof(szEucKr));		// 魹ｽﾑｱ魹ｽ ﾄｳ魹ｽ魹ｽ魹ｽﾍｼ魹ｽ 魹ｽ魹ｽﾈｯ

					strcpy(structBinary.szDevName, szEucKr);



					structBinary.nValue = m_Status.abIOOutputStatus[i];	// Value (Int)



					structBinary.cUnit = (char)99;							// Unit



					structBinary.fMinValue = 0.f;							// Min Value

					structBinary.fMaxValue = 0.f;							// Min Value



					structBinary.cRS = ASCII_RS;



					vSend.push_back(structBinary);

				}





			}



			

			char* pSendProtocol = nullptr;

			int nHeadTailSize = 0, nCurPos = 0;

			char* pCurPt = nullptr;

			

			pSendProtocol = new char[(sizeof(StructBinarySend) * vSend.size()) + 32];

			memset(pSendProtocol, 0, sizeof(StructBinarySend));





			////////////////////////////////

			// Header

			sprintf(pSendProtocol, "%cms.res.%s%c", ASCII_STX, strJobType.c_str(), ASCII_GS);

			pCurPt = pSendProtocol + strlen(pSendProtocol);



			nHeadTailSize = (int)strlen(pSendProtocol) + 1;		// Header 魹ｽ魹ｽ魹ｽ魹ｽ + 魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ ETX



			// 魹ｽ魹ｽ魹ｽ魹ｽ 魹ｽ魹ｽ魹ｽ魹ｽ(魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ 魹ｽﾛｼ魹ｽ魹ｽ魹ｽ 魹ｽ魹ｽ)

			for (i = 0; i < (int)vSend.size(); i++)

			{

				memcpy(pCurPt, &vSend[i], sizeof(StructBinarySend));

				pCurPt += sizeof(StructBinarySend);

			}







			/////////////////////////////

			// Tail

			//sprintf(pCurPt, "%c", ASCII_ETX);			// ETX

			*pCurPt = (char)ASCII_ETX;

			pCurPt++;





			/////////////////////////////

			// CRC

			nCurPos = nHeadTailSize + (int)(sizeof(StructBinarySend) * vSend.size());

			usCRC = gpParam->getCRC16(0, (unsigned char*)pSendProtocol, nCurPos);



			pSendProtocol[nCurPos] = (unsigned char)(usCRC & 0x00FF);						// CRC(LO)

			pSendProtocol[nCurPos + 1] = (unsigned char)((usCRC & 0xFF00) >> 8);				// CRC(HI)

				



			pWorker->send(pSendProtocol, nCurPos + 2);		// CRC 2Byte魹ｽﾟｰ魹ｽ魹ｽﾘｼ魹ｽ 魹ｽ魹ｽ魹ｽ魹ｽ





			delete[] pSendProtocol;



			nRtn = 1;

		}

		else

			nRtn = 0;

	}



	if (nRtn == 0)

	{

		sprintf(sProtocol, "%cb%cerror%c%d%c", ASCII_STX, ASCII_GS, ASCII_US, 1, ASCII_ETX);	// Invalid Protocol

		pWorker->send(sProtocol, strlen(sProtocol));

	}

}

void CStatusHandler::runMESThread(void* arg)

{

	const int nTerm = 100;

	//int nCount = 0, nExecCount, nFailCount = 0;

	int nSleepMs = gpParam->getIntOpt("MES_CONN_SEND_FREQ");

	bool bExecute = gpParam->getIntOpt("MES_CONN_SEND_URL_USE");

	std::string strURL, strResult, strData;

	int nSendType = 1;		// 0:GET, 1:POST (魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ POST魹ｽ魹ｽ魹ｽ魹ｽ. 魹ｽ魹ｽ魹ｽ魹ｽ 魹ｽﾄｶ魹ｽ魹ｽ魹ｽ魹ｽ 魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ 魹ｽ魹ｽ魹ｽ魹ｽ)

	bool bExecSend = false, bCheckDate = false;





	struct timespec timeLast, timeCur;

	long lElapsedTime = 0;

	



	clock_gettime(CLOCK_MONOTONIC, &timeLast);

	



	while (m_bMESThreadRun)

	{

		nSleepMs = gpParam->getIntOpt("MES_CONN_SEND_FREQ");		// MSManager魹ｽ魹ｽ 魹ｽ魹ｽ魹ｽ魹ｽ 魹ｽﾟｰ魹ｽ魹ｽ魹ｽ 魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ 魹ｽ魹ｽ 魹ｽ魹ｽ魹ｽ魹ｽ魹ｽﾇｷ魹ｽ 魹ｽ魹ｽ魹ｽ ﾃｼﾅｩ魹ｽﾑｴ魹ｽ.

		bCheckDate = true;



		if (!bExecute)

		{

			std::this_thread::sleep_for(std::chrono::milliseconds(1000));

			continue;

		}





		clock_gettime(CLOCK_MONOTONIC, &timeCur);



		lElapsedTime = 1000000000 * (timeCur.tv_sec - timeLast.tv_sec) + (timeCur.tv_nsec - timeLast.tv_nsec);



		if (lElapsedTime / 1000000 >= nSleepMs)

		{

			timeLast = timeCur;



			bExecSend = true;

		}

void CStatusHandler::sendLineMessage(int nPortNo, int nSlave, int nTokenIndex)

	else if(nTokenIndex == 1)

	else if (nTokenIndex == 2)

		strcpy(szTokenKey, "MES_LINE_TOKEN3");





	tmStart = m_tmSendLine[nPortNo][nSlave];

	if (strcmp(gpParam->getStrOpt(szTokenKey), "") == 0)

	FILE* stream;

	char buf[16384] = { 0, }, szMsg[512] = { 0, }, szUTF8[512] = { 0, }, szDevNmKey[64] = { 0, }, szMinKey[64] = { 0, }, szMaxKey[64] = { 0, };

	sprintf(szMaxKey, "SERIAL_MAXVALUE_%d_%d", nPortNo, nSlave);



	CEVUtils::evEuckrToUtf8("魹ｽ魹ｽﾄ｡魹ｽ魹ｽ 魹ｽ魹ｽ魹ｽ魹ｽ 魹ｽ魹ｽ魹ｽﾄ｡魹ｽ魹ｽ 魹ｽ魹ｽ魹ｽ讝ｵ魹ｽ魹ｽ魹ｽﾏｴ魹ｽ. 魹ｽ魹ｽ魹ｽ邁ｪ", szUTF8, sizeof(szUTF8));

		gpParam->getStrOpt("DEV_NAME"),

		gpParam->getStrOpt(szDevNmKey),

		szUTF8,

		gpParam->getDblOpt(szMinKey),

		gpParam->getDblOpt(szMaxKey)

		);



	//CEVUtils::evEuckrToUtf8(szMsg, szUTF8, sizeof(szUTF8));





	strCmd.reserve(512);



	/*

	-F 'message=TEST Message 魹ｽﾑｱ魹ｽ'

	strCmd.append("curl --connect-timeout 5 -X POST https://notify-api.line.me/api/notify ");

	strCmd.append("-H 'Authorization:Bearer ")

		.append(gpParam->getStrOpt(szTokenKey))

		.append("' ");

	strCmd.append("-F 'message=")

		.append(szMsg)

		.append("'");





	writeSystemInfo("%s", strCmd.c_str());

	stream = popen(strCmd.c_str(), "r");

StructResultVariable CStatusHandler::getResultVariable(int nIndex)

{

	StructResultVariable vars;



	if (nIndex >= RESULT_VARIABLE_CNT)

	{



		vars.nValue = 0;

		vars.fValue = 0;

		memset(vars.szData, 0, sizeof(vars.szData));

		memset(vars.szUnit, 0, sizeof(vars.szUnit));



		return vars;

	}



	m_mutexResultVar.lock();

	vars.fValue = m_vResultVariable[nIndex].fValue;

	vars.nValue = m_vResultVariable[nIndex].nValue;

	strcpy(vars.szData, m_vResultVariable[nIndex].szData);

	strcpy(vars.szUnit, m_vResultVariable[nIndex].szUnit);

	m_mutexResultVar.unlock();



	return vars;



}



void CStatusHandler::setResultVariable(int nIndex, StructResultVariable vars)

{

	if (nIndex >= RESULT_VARIABLE_CNT)

		return;





	m_mutexResultVar.lock();



	m_vResultVariable[nIndex].fValue = vars.fValue;

	m_vResultVariable[nIndex].nValue = vars.nValue;

	strcpy(m_vResultVariable[nIndex].szData, vars.szData);

	strcpy(m_vResultVariable[nIndex].szUnit, vars.szUnit);



	m_mutexResultVar.unlock();

}

*/

unsigned int CNetworkHandler::getNetMaskBitCount(std::string strNetMask)

{

	std::string strTok;

	std::stringstream sstream(strNetMask);

		

	unsigned int uiMask = 0, uiItem = 0;

	int nShifter[4] = { 24, 16, 8, 0 }, nCur = 0;

	int nOnCount = 0;



	while (getline(sstream, strTok, '.'))

	{

		uiItem = (unsigned int)atoi(strTok.c_str());

		uiMask |= (uiItem << nShifter[nCur]);



		nCur++;

	}





	for (int i = 0; i < 32; i++)

	{

		if (uiMask & 0x01)

			nOnCount++;



		uiMask = uiMask >> 1;

	}



	return nOnCount;

}

bool CNetworkHandler::getGateway(char* szGateway)

{

	std::string strLine, strValue;

	std::string::size_type stFind = 0;

	size_t stLen = 0, i;



	fp = fopen("/etc/dhcpcd.conf", "r");



	if (fp == nullptr)

	{

		writeSystemError("[CNetworkHandler] /etc/dhcpcd.conf file can't open.");

		return false;

	}



	while (!feof(fp))

	{

		memset(buf, 0, sizeof(buf));

		fgets(buf, sizeof(buf), fp);



		if (buf == nullptr)

			break;



		if (buf[0] == '#' || buf[0] == 0x0A)

			continue;



		stLen = strlen(buf);

		for (i = 0; i < stLen; i++)

		{

			if (buf[i] == 0x0A || buf[i] == 0x0D)

				buf[i] = 0x0;

		}



		strLine = buf;



		if (strLine.find("static routers", 0) != std::string::npos)

			if (stFind != std::string::npos)

			{

				break;

			}

		}

	}





	fclose(fp);



	return true;

}

void CEVServerSocket::stop()

{



	writeSystemInfo("[CEVServerSocket] Server Socket stopping...");



	m_bRunning = false;

		

	if (m_vecWorker.size() > 0)

	{

		writeSystemDebug("[CEVServerSocket] Server Socket Worker thread exist. Worker stopping...");



		for (int i = 0; i < (int)m_vecWorker.size(); i++)

		{

			writeSystemDebug("	[CEVServerSocket] Worker(Thread ID : %d) stopping. ", m_vecWorker[i]->getThreadID());



			m_vecWorker[i]->stop();

			std::this_thread::sleep_for(std::chrono::milliseconds(10));

			delete m_vecWorker[i];

		}



		writeSystemDebug("[CEVServerSocket] All Worker stopped.");

	}





	if (m_pThreadListener != NULL)

	{

		



		writeSystemDebug("[CEVServerSocket] Server Socket  closing...");

		close(m_nServerSocketFd);



		writeSystemDebug("[CEVServerSocket] Server Socket Listner thread waiting...");

		m_pThreadListener->join();



		writeSystemDebug("[CEVServerSocket] Server Socket Listner thread deleting...");

		delete m_pThreadListener;

		m_pThreadListener = NULL;

	}



	writeSystemInfo("[CEVServerSocket] Server Socket stopped.");

}

	if (bind(m_nServerSocketFd, (struct sockaddr*)&m_ServerAddr, sizeof(m_ServerAddr)) == -1)

	{

		writeSystemError("[CEVServerSocket] bind() error.");

		return;

	}





	// Listen

	if (listen(m_nServerSocketFd, BACKLOG_COUNT) == -1)

	{	

		writeSystemError("[CEVServerSocket] listen() error.");

		return;

	}





	writeSystemInfo("[CEVServerSocket] Server Socket start listening.(Port:%d)", ntohs(m_ServerAddr.sin_port));



	// NonBlocking魹ｽ魹ｽ魹ｽ魹ｽ accept魹ｽ魹ｽ 魹ｽ魹ｽ魹ｽ魹ｽ魹ｽﾑｴ魹ｽ.(Blocking魹ｽ魹ｽ魹ｽ魹ｽ 魹ｽ魹ｽ魹ｽ魹ｽ 魹ｽ魹ｽ魹ｽ stop()魹ｽ魹ｽ魹ｽ魹ｽ 魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ 魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ 魹ｽ魹ｽ accept魹ｽ魹ｽ魹ｽ魹ｽ 魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽﾃｼ魹ｽ魹ｽ魹ｽ 魹ｽ魹ｽ魹ｽ魹ｽ)

	int nFlags = fcntl(m_nServerSocketFd, F_GETFL, 0);

	fcntl(m_nServerSocketFd, F_SETFL, O_NONBLOCK | nFlags);



	while (m_bRunning)

	{	



		// Accept

		struct sockaddr_in clientAddr;

		socklen_t clientLen = sizeof(clientAddr);

		int nClientFd = -1;

				

		nClientFd = accept(m_nServerSocketFd, (struct sockaddr*)&clientAddr, &clientLen);



		// 魹ｽﾖｴ魹ｽ ﾅｬ魹ｽ魹ｽ魹ｽﾌｾ魹ｽﾆｮ 魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ 魹ｽﾊｰ魹ｽ魹ｽﾟｴﾙｸ魹ｽ 魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ 魹ｽﾙｷ魹ｽ 魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ.

		if (m_vecWorker.size() >= (size_t)m_nMaxClient)

		{

			close(nClientFd);

			nClientFd = -1;

		}





		//////////////////////////////////////////

		// ﾅｬ魹ｽ魹ｽ魹ｽﾌｾ魹ｽﾆｮ 魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ 魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ 魹ｽ魹ｽ魹ｽ魹ｽ

		if (nClientFd != -1)

		{

			CEVServerSocketWorker* workerThread = new CEVServerSocketWorker(nClientFd, clientAddr, this);



			if (workerThread != NULL)

			{

				m_mutexList.lock();



				workerThread->start();

				std::this_thread::sleep_for(std::chrono::milliseconds(10));

				m_vecWorker.push_back(workerThread);



				writeSystemInfo("[CEVServerSocket] Current connected client(s) : %d", m_vecWorker.size());



				m_mutexList.unlock();

			}

		}

	





		// ﾅｬ魹ｽ魹ｽ魹ｽﾌｾ魹ｽﾆｮ 魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ Worker魹ｽ魹ｽ 魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽﾃｻ魹ｽ魹ｽ 魹ｽ魹ｽ CEVServerSocketWorker魹ｽ魹ｽ 魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ 魹ｽ魹ｽ 魹ｽﾗｸ魹ｽ魹ｽ魹ｽ 魹ｽ魹ｽ魹ｽ魹ｽﾃｳ魹ｽ魹ｽ魹ｽﾑｴ魹ｽ.

		this->removeWorkerFromList();





		std::this_thread::sleep_for(std::chrono::milliseconds(50));

	}







	writeSystemInfo("[CEVServerSocket] Server Socket listening stopped.");



	//close(m_nServerSocketFd);



}







void CEVServerSocket::requestRemoveWorkerFromList(CEVServerSocketWorker* pWorker)

{

	m_mutexList.lock();



	m_vecReqRemoveList.push_back(pWorker);



	m_mutexList.unlock();

}



/*

CEVServerSocketWorker魹ｽ魹ｽ 魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ 魹ｽﾔｼ魹ｽ startComm()魹ｽ魹ｽ 魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ Worker List Vector魹ｽ魹ｽ魹ｽ魹ｽ 魹ｽ魹ｽ魹ｽﾅｵﾇｱ魹ｽ 魹ｽ魹ｽ魹ｽ魹ｽ Worker魹ｽﾚｽ魹ｽ魹ｽ魹ｽ ﾈ｣魹ｽ魹ｽ魹ｽﾑｴ魹ｽ.

*/

void CEVServerSocket::removeWorkerFromList()

{

	if (m_vecReqRemoveList.size() > 0)

	{	// ﾅｬ魹ｽ魹ｽ魹ｽﾌｾ魹ｽﾆｮ 魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ 魹ｽ魹ｽ魹ｽ魹ｽ 魹ｽﾚｽ魹ｽ魹ｽ魹ｽ 魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ 魹ｽﾞｶ魹ｽ魹ｽ 魹ｽ魹ｽﾃｻ魹ｽ魹ｽ 魹ｽ魹ｽ CEVServerSocketWorker魹ｽ魹ｽ 魹ｽﾖｴﾙｸ魹ｽ 魹ｽﾘｴ魹ｽ Worker魹ｽ魹ｽ 魹ｽ魹ｽ魹ｽ魹ｽﾃｳ魹ｽ魹ｽ魹ｽﾑｴ魹ｽ.

		m_mutexList.lock();



		std::this_thread::sleep_for(std::chrono::milliseconds(20));



		for (vector<CEVServerSocketWorker*>::iterator itReq = m_vecReqRemoveList.begin(); itReq != m_vecReqRemoveList.end(); ++itReq)

		{

			for (vector<CEVServerSocketWorker*>::iterator it = m_vecWorker.begin(); it != m_vecWorker.end(); ++it)

			{

				if (*it == *itReq)

				{

					if ((*it)->isRunning())

						(*it)->stop();



					m_vecWorker.erase(it);



					delete *itReq;



					break;

				}

			}



		}



		m_vecReqRemoveList.clear();



		m_mutexList.unlock();





		writeSystemInfo("[CEVServerSocket] Current Worker(Connected client) Count : %d", m_vecWorker.size());



	}



}







void CEVServerSocket::sendAll(const char* pData, size_t nLen)

{

	for (vector<CEVServerSocketWorker*>::iterator it = m_vecWorker.begin(); it != m_vecWorker.end(); ++it)

	{

		if (!(*it)->isRunning())

			continue;

		

		ssize_t nRtn = write((*it)->getCliendFd(), pData, nLen);

		

		if (nRtn < 0)

		{

			writeSystemError("[CEVServerSocket] Writing failed.(Client FD:%d)", (*it)->getCliendFd());

		}

	}

}

	{	

		callbackReceiveData(pWorker, nLen);

	}

}

{

		return !std::isspace(ch);

		}));

}

{

	s.erase(std::find_if(s.rbegin(), s.rend(), [](unsigned char ch) {

		}).base(), s.end());

}
